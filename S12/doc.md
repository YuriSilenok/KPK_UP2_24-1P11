## ER-диаграмма

```mermaid
erDiagram
    group {
        int id PK
        string name
        bool is_active
    }
    
    discipline {
        int id PK
        string name
        bool is_active
    }
    
    semester {
        int id PK
        int semester_number
        string academic_year
        bool is_active
    }
    
    curriculum {
        int id PK
        int group_id FK
        int discipline_id FK
        int semester_id FK
        int theory_hours
        int practice_hours
        string assessment_form
        bool is_active
    }
    
    // Связи приведены к формату "id → foreign_key"
    id --> group_id : "group"
    id --> discipline_id : "discipline"
    id --> semester_id : "semester"
